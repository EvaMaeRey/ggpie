#' @return

#' @examples
defaults_pie <- function(...){
  
  list(
    ggplot2::coord_polar(theta = "y", ...),
    ggplot2::theme_void(),
    ggplot2::aes(x = 0) # hacky; grammar problem
  )
  
}

#' Title
#'
#' @param data 
#'
#' @return
